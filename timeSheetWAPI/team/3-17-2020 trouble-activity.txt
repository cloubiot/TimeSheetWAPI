CREATE TABLE `timesheet`.`trouble_activity` (
  `ID` INT(11) NOT NULL,
  `ISSUE_ID` INT(11) NULL DEFAULT NULL,
  `DESCRIPTION` VARCHAR(255) NULL DEFAULT NULL,
  `COMMENTS` LONGTEXT NULL DEFAULT NULL,
  `STATUS_ID` INT(11) NULL DEFAULT '2',
  `CREATED_DATE` DATETIME NULL DEFAULT NULL,
  `LAST_MOD_USER` VARCHAR(45) NULL DEFAULT NULL,
  `LAST_MOD_DATE` DATETIME NULL DEFAULT NULL,
  `ROLE_ID` INT(11) NULL DEFAULT NULL,
  `MAIL_NOTES` TINYINT(4) NULL DEFAULT '0',
  PRIMARY KEY (`ID`));
